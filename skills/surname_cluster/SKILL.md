```python
def get_surname_samples(surname, data):
```

```python
def determine_cluster_type(samples):
    """
    –û–ø—Ä–µ–¥–µ–ª—è–µ—Ç, —è–≤–ª—è–µ—Ç—Å—è –ª–∏ —Ñ–∞–º–∏–ª–∏—è –º–æ–Ω–æ- –∏–ª–∏ –ø–æ–ª–∏–≥–∞–ø–ª–æ–≥—Ä—É–ø–ø–Ω–æ–π

    """
```

```python
    result = {
        'type': 'POLY',
        'haplogroups': {},
        'dominant_haplogroup': None,

    # –°—á–∏—Ç–∞–µ–º —á–∞—Å—Ç–æ—Ç—ã –≥–∞–ø–ª–æ–≥—Ä—É–ø–ø
    hap_counts = {}
    for s in samples:

        hap_counts[hap] = hap_counts.get(hap, 0) + 1

    # –û–ø—Ä–µ–¥–µ–ª—è–µ–º –¥–æ–º–∏–Ω–∞–Ω—Ç–Ω—É—é –ª–∏–Ω–∏—é

        else:
            result['minor_haplogroups'].append({'haplogroup': hap, 'count': count})

```
